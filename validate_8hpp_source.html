51 template <typename T>
52 struct validate_range {
53  T from = std::numeric_limits<T>::min();
54  T to = std::numeric_limits<T>::max();
55  std::optional<T> step = std::nullopt;
56  double step_threshold = 1e-3;
57 
58  constexpr Result<T> operator()(T value, std::string const& name) const {
59  if (value < from || value > to) {
60  return tl::make_unexpected(OutOfRange(fmt::format(
61  "{}: {} is outside of the range [{}, {}]", name, value, from, to)));
62  }
63 
64  if (step) {
65  double const step_value = static_cast<double>(step.value());
66  double const ratio = static_cast<double>(value - from) / step_value;
67  double const distance = fabs(ratio - round(ratio));
68 
69  if (distance > step_threshold) {
70  return tl::make_unexpected(OutOfRange(
71  fmt::format("{}: {} is {} away from the nearest valid step", name,
72  value, distance)));
73  }
74  }
75 
76  return value;
77  }
78 };
