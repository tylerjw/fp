33 fp::Result<double> divide_4_by(double x) {
34  if (x == 0.0) {

45 fp::Result<double> do_work() {
46  auto const a = fp::make_result(2.4) | f1 | f2;
47  auto const b = fp::make_result(0.0) | f3 | f4;
48 
49  if (auto const error = fp::maybe_error(a, b); error) {
50  return tl::make_unexpected(*error);
51  }
52 
53  return *a + *b;
54 }
