192 template <typename E, typename... Args>
193 constexpr std::optional<E> maybe_error(tl::expected<Args, E>... args) {
194  auto maybe = std::optional<E>{std::nullopt};
195  (
196  [&](auto& exp) {
197  if (maybe.has_value()) return;
198  if (has_error(exp)) maybe = exp.error();
199  }(args),
200  ...);
201  return maybe;
202 }
